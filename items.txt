            <div class="enlightenment-text lore">The Formless One of Enlightenment is never named in the game's dialogue — only implied. Where Caligo is described as existing <em>the way weather exists</em>, Enlightenment is its inverse: a presence that does not corrupt but illuminates, and not always kindly. The Book is the only confirmed physical artifact of Enlightenment's influence. It was not created by scholars or the Church. The Ancient Paper from which it is crafted predates the Theocratic capital by four hundred years. The Librarian Scholar, in his extended dialogue, says only: <em>"There are things that want to be known. The book is one of them."</em> He does not explain further. He never explains further. That is the whole of what Enlightenment gives: the information, without the comfort of understanding it.</div>
            <div class="enlightenment-section-label">Notes</div>
            <div class="enlightenment-text">The Book of Enlightenment is the game's rarest crafted item. Its relationship to Caligo is oppositional but not adversarial — Caligo corrupts through absence of meaning; Enlightenment overwhelms through excess of it. Both result in the same risk: a mind that cannot hold what it has been given.</div>

const filterBtns = document.querySelectorAll('.filter-btn');
const itemCountEl = document.getElementById('item-count');

function updateCount() {
  const activeFilter = document.querySelector('.filter-btn.active')?.dataset.filter || 'all';
  const all = document.querySelectorAll('[data-category]');
  let visible = 0;
  all.forEach(el => {
    const show = activeFilter === 'all' || el.dataset.category === activeFilter;
    el.style.display = show ? '' : 'none';
    if (show) visible++;
  });
  itemCountEl.textContent = `${visible} items`;
  document.querySelectorAll('.category-group').forEach(group => {
    const anyVisible = group.querySelectorAll('[data-category]:not([style*="none"])').length > 0;
    group.style.display = anyVisible ? '' : 'none';
  });
}

filterBtns.forEach(btn => {
  btn.addEventListener('click', () => {
    filterBtns.forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    updateCount();
  });
});
updateCount();

// ── SEARCH ──────────────────────────────
const itemIndex = [];
document.querySelectorAll('[data-category]').forEach(card => {
  const name = card.querySelector('.item-name, .fragment-card-name, .enlightenment-card-name');
  const texts = card.querySelectorAll('.item-section-text, .fragment-section-text, .enlightenment-text');
  if (name) {
    itemIndex.push({
      name: name.textContent.trim(),
      text: Array.from(texts).map(t => t.textContent).join(' ').toLowerCase(),
      el: card
    });
  }
});

const inputEl = document.getElementById('search-input');
const overlayEl = document.getElementById('search-results-overlay');

inputEl.addEventListener('input', () => {
  const q = inputEl.value.trim().toLowerCase();
  overlayEl.innerHTML = '';
  if (!q || q.length < 2) { overlayEl.classList.remove('visible'); return; }
  const results = itemIndex.filter(i => i.name.toLowerCase().includes(q) || i.text.includes(q)).slice(0, 10);
  if (!results.length) {
    overlayEl.innerHTML = '<div class="search-no-result">No items found for "' + q + '"</div>';
  } else {
    results.forEach(item => {
      const div = document.createElement('div');
      div.className = 'search-result-item';
      div.innerHTML = `<div class="sr-title">${highlightMatch(item.name, q)}</div><div class="sr-excerpt">Click to scroll to item</div>`;
      div.onclick = () => {
        overlayEl.classList.remove('visible');
        item.el.scrollIntoView({ behavior: 'smooth', block: 'center' });
        item.el.style.outline = '1px solid var(--gold)';
        setTimeout(() => { item.el.style.outline = ''; }, 2000);
      };
      overlayEl.appendChild(div);
    });
  }
  overlayEl.classList.add('visible');
});

inputEl.addEventListener('keydown', e => { if (e.key === 'Escape') { overlayEl.classList.remove('visible'); inputEl.blur(); } });
document.addEventListener('click', e => { if (!e.target.closest('.search-wrap') && !e.target.closest('#search-results-overlay')) overlayEl.classList.remove('visible'); });

function highlightMatch(text, q) {
  const i = text.toLowerCase().indexOf(q);
  if (i < 0) return text;
  return text.slice(0, i) + '<mark class="search-highlight">' + text.slice(i, i + q.length) + '</mark>' + text.slice(i + q.length);
}

  entries.forEach(e => {
    if (e.isIntersecting) { e.target.style.opacity = '1'; e.target.style.transform = 'translateY(0)'; }
  });
}, { threshold: 0.06, rootMargin: '0px 0px -30px 0px' });

  el.style.transition = 'opacity 0.45s ease, transform 0.45s ease';
  observer.observe(el);
});
