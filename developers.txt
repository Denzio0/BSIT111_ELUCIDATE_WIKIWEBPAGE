];

function escapeHtml(str) {
  return str.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
function highlightMatch(text, q) {
  const idx = text.toLowerCase().indexOf(q);
  if (idx < 0) return escapeHtml(text);
  return escapeHtml(text.slice(0, idx)) +
    '<mark class="search-highlight">' + escapeHtml(text.slice(idx, idx + q.length)) + '</mark>' +
    escapeHtml(text.slice(idx + q.length));
}
function performSearch() {
  const q = inputEl.value.trim().toLowerCase();
  overlayEl.innerHTML = '';
  if (!q || q.length < 2) { overlayEl.classList.remove('visible'); return; }
  const results = searchIndex.filter(item =>
    item.title.toLowerCase().includes(q) ||
    item.excerpt.toLowerCase().includes(q) ||
    item.section.toLowerCase().includes(q)
  ).slice(0, 10);
  if (!results.length) {
    overlayEl.innerHTML = '<div class="search-no-result">No results for "' + escapeHtml(q) + '"</div>';
  } else {
    results.forEach(item => {
      const div = document.createElement('div');
      div.className = 'search-result-item';
      div.innerHTML = `<div class="sr-title">${highlightMatch(item.title, q)} <span style="font-size:9px;color:var(--text-dim);margin-left:6px;">${item.section}</span></div><div class="sr-excerpt">${highlightMatch(item.excerpt.slice(0,80) + '…', q)}</div>`;
      div.onclick = () => { if (item.href.startsWith('#')) { const el = document.querySelector(item.href); if (el) el.scrollIntoView({ behavior: 'smooth' }); } else window.location.href = item.href; overlayEl.classList.remove('visible'); };
      overlayEl.appendChild(div);
    });
  }
  overlayEl.classList.add('visible');
}
inputEl.addEventListener('input', performSearch);
inputEl.addEventListener('keydown', e => { if (e.key === 'Escape') { overlayEl.classList.remove('visible'); inputEl.blur(); } });
document.addEventListener('click', e => { if (!e.target.closest('.search-wrap') && !e.target.closest('#search-results-overlay')) overlayEl.classList.remove('visible'); });

const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => { if (entry.isIntersecting) entry.target.classList.add('visible'); });
}, { threshold: 0.08, rootMargin: '0px 0px -30px 0px' });
document.querySelectorAll('.reveal').forEach(el => observer.observe(el));
</script>
